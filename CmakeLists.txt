cmake_minimum_required(VERSION 3.15)
project(Lab_2 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 26)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/bin)

enable_testing() 

add_executable(lab2_app src/main.cpp)

#add_subdirectory(lib/PDCurses)
add_subdirectory(lib/googletest)
add_subdirectory(lib/ftxui)

add_subdirectory(src/core)
add_subdirectory(src/sequences)
add_subdirectory(src/sequences/SequenceArray)
add_subdirectory(src/sequences/SequenceList)

target_link_libraries(lab2_app PRIVATE
    ftxui::dom
    ftxui::screen
    ftxui::component
    lab2_core
    lab2_sequences
    lab2_seqarray
    lab2_seqlist
)

target_compile_definitions(lab2_app PRIVATE _WIN32_WINNT=0x0600)

add_subdirectory(tests)
